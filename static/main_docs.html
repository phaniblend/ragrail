Create reusable logic with custom hooks to keep components clean.

function useLocalStorage(key, initialValue) { const [value, setValue] = useState(() => { const item = localStorage.getItem(key); return item ? JSON.parse(item) : initialValue; }); const setStoredValue = useCallback((newValue) => { setValue(newValue); localStorage.setItem(key, JSON.stringify(newValue)); }, [key]); return [value, setStoredValue]; }

Custom hooks enable you to extract component logic into reusable functions.

#!/usr/bin/env node const fs = require('fs'); const path = require('path'); class FileConsolidator { constructor(rootPath) { this.rootPath = path.resolve(rootPath); this.consolidatedCode = []; this.filesIncluded = 0; } consolidate() { console.log('üöÄ Consolidating project files...'); this.scanDirectory(this.rootPath); const header = `PROJECT CONSOLIDATION Generated: ${new Date().toISOString()} Files: ${this.filesIncluded} `; const content = header + this.consolidatedCode.join('\n'); fs.writeFileSync('consolidated-project.txt', content); console.log(`‚úÖ Saved consolidated-project.txt (${this.filesIncluded} files)`); return content; } scanDirectory(dirPath) { try { const entries = fs.readdirSync(dirPath); for (const entry of entries) { const fullPath = path.join(dirPath, entry); if (this.shouldIgnore(fullPath)) continue; const stats = fs.statSync(fullPath); if (stats.isDirectory()) { this.scanDirectory(fullPath); } else if (this.isCodeFile(fullPath)) { this.processFile(fullPath); } } } catch (error) { console.warn(`Warning: Could not scan ${dirPath}`); } } shouldIgnore(filePath) { const ignored = ['node_modules', '.git', 'dist', 'build', '.next']; return ignored.some(pattern => filePath.includes(pattern)); } isCodeFile(filePath) { const ext = path.extname(filePath).toLowerCase(); return ['.js', '.jsx', '.ts', '.tsx', '.json', '.md'].includes(ext); } processFile(filePath) { try { const content = fs.readFileSync(filePath, 'utf8'); const relativePath = path.relative(this.rootPath, filePath); const section = `${'='.repeat(80)} FILE: ${relativePath} ${'='.repeat(80)} ${content} `; this.consolidatedCode.push(section); this.filesIncluded++; console.log(`‚úì Added ${relativePath}`); } catch (error) { console.warn(`Warning: Could not process ${filePath}`); } } } if (require.main === module) { const projectPath = process.argv[2] || process.cwd(); try { const consolidator = new FileConsolidator(projectPath); consolidator.consolidate(); } catch (error) { console.error(`‚ùå Error: ${error.message}`); process.exit(1); } }
